def self.find(board_id)
  api_call = ONE_BOARD.gsub("{boardID}", board_id.to_s)
  url      = "#{LeanKitKanban::Config.uri}#{api_call}"
  response = get(url, LeanKitKanban::Config.basic_auth_hash)
  parse_body(response.body)
end
